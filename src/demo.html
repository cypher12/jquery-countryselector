jQuery Country Selector - Demo

Quick setup
<select data-role="country-selector"></select>
<select id="quick-setup"></select>
$("#quick-setup").countrySelector();
With initial value
<select value="FRA" data-role="country-selector"></select>
<select id="with-initial-value"></select>
$("#with-initial-value").countrySelector({ value: 'FRA' });
Using ISO 3166-1 alpha-3 code (default)
<select value="CHN" data-role="country-selector" data-code-mode="alpha3"></select>
<select id="alpha3"></select>
$("#alpha3").countrySelector({ value: "CHN", codeMode: "alpha3" });
Using ISO 3166-1 alpha-2 code
<select value="US" data-role="country-selector" data-code-mode="alpha2"></select>
<select id="alpha2"></select>
$("#alpha2").countrySelector({ value: "US", codeMode: "alpha2" });
Using ISO 3166-1 numeric code
<select value="40" data-role="country-selector" data-code-mode="numeric"></select>
<select id="numeric"></select>
$("#numeric").countrySelector({ value: 40, codeMode: "numeric" });
Specify available countries
<select data-role="country-selector" data-value-set="FRA ITA CHE DEU"></select>
<select id="value-set"></select>
$("#value-set").countrySelector({ valueSet: ["FRA", "ITA", "CHE", "DEU"] });
Showing country flags (experimental)
<select data-role="country-selector" data-show-flag="true"></select>
<select id="flag"></select>
$("#flag").countrySelector({ showFlag: true });
