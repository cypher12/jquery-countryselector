jQuery Country Selector - Samples

Initialization

            ('#country').countrySelector({
                countries: ['FRA', 'DEU', 'ITA', 'FIN', 'SWE'],
                onSelected: function (country) {
                    $('#text').text('Selected country: ' + country);
                }
            });

Get function

            $('#get-selected-country').click(function () {
                var country = $('#country').countrySelector('value');
                alert('Selected country is: ' + country);
            });

Set function

            $('#set-FRA').click(function () {
                $('#country').countrySelector('value', 'FRA');
            });
