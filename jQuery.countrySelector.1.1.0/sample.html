jQuery Country Selector - Samples

Div constructor

            ('#country').countrySelector({
                countries: ['FRA', 'DEU', 'ITA', 'FIN', 'SWE', 'CHN', 'JPN', 'KOR', 'GBR'],
                onSelected: function (country, fullname) {
                    $('#text').text('Selected country: ' + country + ' (' + fullname + ')');
                }
            });
